async, await:

что бы сделать функцию асинзронной, нужно перед словом function написать
async;

вот как примерно выглядит базовый синтаксис async/await:

    async function loadUserData() {
        try {
            //отправляем запрос (по умолчанию GET)
            const response = await fetch('https://example.com');

            //проверяем статус ответа (200-299)
            if (!response.ok) {
                throw new Error(`Ошибка сервера: ${response.status});
            }

            //декодируем JSON в обычный JS
            const data = await response.json();
            console.log(data);
        }
        catch(error) {
            //обрабатываем сетевые ошибки
            console.log('Ошибка запроса: ', error)
        }
    }

вот так выглядит отправка данных (POST-запрос):

    fetch('https://example.com', {
        method: 'POST' //указываем HTTP-метод
        headers: {
            'Content-Type': 'application/json' //указываем серверу, что отправляем JSON
        },
        body: JSON.stringify({ name: 'Карбел', role: 'admin' }) //превращаем объект в строку
    });

есть 3 главных метода обработки ответа:
    - response.json() - парсит тело ответа в формате JSON в объект JS;
    - response.text() - возвращает ответ в виде обычной текстовой строки;
    - response.blob() - считывает данные как бинарный файл (картинка, документ);

бывают моменты, когда нужно отменять запрос либо через какое-то время, либо
при новом запросе:

    отмена запроса по таймеру:

        async function fetchWithTimeout(url, timeoutMs = 5000) {
            //создаём контроллер
            const controller = new abortController();
            const { signal } = controller;

            //запускаем таймер на отмену
            const timer = setTimeout(() => controller.abort(), timeoutMs);

            try {
                //передаём сигнал в опции fetch
                const response = await fetch(url, { signal });
                return await response.json();
            }
            catch (error) {
                //проверяем был ли запрос отменён вручную / по таймеру
                if (error.name === 'abortError') {
                    console.warn('Запрос отменён: превышено время ожидания!');
                }
                else {
                    console.error('Другая ошибка: ', error);
                }
            }
            finally {
                clearTimeout(timer); //очищаем таймер в любом случае
            }
        }

    отмена старого запроса при новом:

        let currentController = null;

        async function loadDataOnClick() {
            //если прошлый запрос ещё идёт - отменяем его
            if (currentController) {
                currentController.abort();
            }

            //создаём новый контроллер для текущего запроса
            currentController = new AbortController();

            try {
                const response = await fetch('/api/heavy-data', { signal: currentController.signal });
                const data = await response.json();
                console.log('Данные успешно загружены: ', data);
            }
            catch (error) {
                if (error.name !=='AbortError') {
                    //игнорируем AbortError, обрабатывая только реальные ошибки
                    console.log('Ошибка загрузки: ', error);
                }
            }
        }

    у AbortController() есть пульт управления - .abort и некий провод - .signal;
    привязывая к fetch контролер таким образом: 
    { signal: AbortController().signal } - указывание браузеру, что ему нужно
    следить за AbortController() и как только будет подан сигнал отмены, то нужно
    сразу прервать выполнение запроса;

    банальные примеры, где такое нужно:
    
        - при вводе слова в поиск: вместо обработки каждого изменения (с, сл, сло)
        каждый новый запрос должен отменяться, ибо может случиться такое, что запрос
        на ту же букву "с" будет идти дольше всего, отчего придёт последним и на финальный
        запрос от пользователя "слово" будет выдан ответ на запрос "с";

        - переключение страниц: пользователь зашёл на страницу профиля, в этот момент
        начали прогружаться данные, но пользователь сразу же перешёл на станицу
        настроек из профиля, поэтому данные страницы уже не актуальны;

        - защита от зависания: если сервер упал, то нужно прерывать бесконечную
        загрузку и выдавать пользователю ошибку serverTimeout;

//сперва мы ждём промис с ответом от сервера
const response = await fetch('something');
//на данный момент получили лишь ответ, что сервер жив (200)
//теперь же нужно сообщить, что надо теперь скачать все данные с запроса
const data = await response.json();
//по завершению этого await в data будет находиться конвертированные в
//js-объект данные, преобразованные из JSON

ВАЖНО! нельзя допускать циклическую зависимость (Circular Dependency):

    это может привести к тому, что переменные могут стать undefined,
    потому что файлы не могут решить, кто из них первый;

    для лечения подобного лучше использовать нейтральную территорию:

        стоит вынести функцию, из-за которой возникает такая циклическая
        зависимость в отдельный файл;

    Мутация vs Иммутабельность

    Мутация:

        когда я пишу hero.inventory.push(item), я беру существующий массив и
        всовываю в него новый элемент;
            плюсы: это быстро и занимает меньше памяти;
            минусы: когда есть 2 части программы, которые следят за этим массивом,
            один из них может не заметить изменений, потому что ссылка на массив
            осталась прежней (тот же самый объект в памяти);

    Иммутабельность:

        вместо изменения старого массива создаётся новый:

            //иммутабельное добавление
            hero.inventory = [...hero.inventory, newItem];

                что происходит: мы говорим: создай новый массив, скопируй туда всё
                из старого и добавь в конец newItem;
                зачем это нужно: в React это единственный способ заставить интерфейс
                обновиться. React сравнивает: oldInventory === newInventory. если я
                сделал push, результат будет true (ссылка-то одна). если сделал через
                spread - результат false, и React понимает - данные изменились,
                нужно перерисовать экран;

    немного иначе про Мутацию и Иммутабельность:

        примитивы всегда иммутабельны:
            строки, числа, булевы значения нельзя изменить внутри памяти:
                let name = 'Карбел';
                //не сработает. строку нельзя мутировать
                name[0] = 'П';
                //перезапись переменной, но старая строка 'Карбел' в памяти не изменилась
                name = 'Маркез';

        объекты и массивы мутабельны по умолчанию:
            они хранятся в памяти по ссылке. переменная указывает на адрес в памяти,
            а не на само значение:
                let user = { name: 'Карбел' };
                //копируется ссылка а не объект
                let admin = user;
                //мутация
                admin.name = 'Маркез';
                //выведет 'Маркез', так как объект 1
                console.log(user.name);

        опасность мутаций (side effects):
            мутации опасны тем, что изменения в одном месте кода могут незаметно
            сломать логику в другом:
                const fruits = ['яблоко', 'апельсин'];

                function visualizer(arr) {
                    //метод .reverse() мутирует исходный массив
                    return arr.reverse();
                }

                const reversed = visualizer(fruits);

                console.log(fruits);
                // ['апельсин', 'яблоко'] - исходный массив испорчен

        чтобы избежать мутаций, используют новые синтаксические возможности js:
            для объектов:
                неправильно (мутация): obj.age = 25;
                правильно (иммутабельно через Spread): const newObj = { ...obj, age: 25 };
                правильно (иммутабельно через Object.assign): const newObj = 
                Object.assign({}, obj, {age: 25});
            для массивов:
                мутирующие методы (избегать): push(), pop(), shift(), unshift(), splice(),
                reverse(), sort();
                иммутабельные методы (использовать): map(), filter(), concat(), slice(),
                reduce(), а также новый метод toReversed() или Spread-оператор [...arr];

            const updatedFruits = [...fruits, 'банан'];

        защита от мутаций Object.freeze():
            метод Object.freeze() делает объект неизменяемым на 1-ом уровне вложенности:
                const config = { host: "localhost" };
                Object.freeze(config);

                config.host = "127.0.0.1"; //ошибка в strict mode или просто проигнорируется

                ВАЖНО! заморозка поверхностная, т.е. если внутри объекта есть другой объект,
                то его всё ещё можно мутировать;
