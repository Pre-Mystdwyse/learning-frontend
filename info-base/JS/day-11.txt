callback-функции:

    обратный вызов - это функция, переданная в другую функцию в качестве
    аргумента, которая затем вызывается по завершению какого-либо действия;
    
    пример:

        //последней выполняется эта функция, которая получила на вход
        //значение переменной name и записала в одноимённую переменную у себя
        function greeting(name) {

            //в итоге, функция вывела текст с введённым в предыдущей функции именем
            alert("Hello " + name);
        }

        //вторым выполняется эта функция. она принимает на вход значение,
        //которое записывается в переменную callback. в данном случае в неё
        //записалась ссылка на функцию greeting
        function processUserInput(callback) {
            //prompt выводит пользователю базовое модальное окно с текстом,
            //полем для ввода и кнопками "ок" и "отмена"
            var name = prompt("Please enter your name.");
            //передаём в переменную callback, в которой хранится ссылка на
            //функцию greeting, переменную name
            callback(name);
        }

        //первым выполняется эта строка. в неё передаётся ссылка
        //на место в памяти функции greeting (переменная может хранить
        //функции тоже)
        processUserInput(greeting);

    есть подобный вариант с асинхронностью:

        //разница лишь в том, что при вызове этой функции, происходит
        //ожидание данных по адресу и только после их загрузки продолжится
        //выполнение всех связанных функций
        async function pageLoader(callback) {
            const data = await fetch("/ru/docs/Glossary/Callback_function");
            callback(data);
        }

        function onPageLoadingFinished(pageData) {
            console.log("Page was sucessfully loaded!");
            console.log("Response: ");
            console.log(pageData);
        }

        pageLoader(onPageLoadingFinished);

    есть более современный вариант использования:

        //функция теперь только загружает и возвращает данные
        async function getPageData() {
            const response = await fetch("/ru/docs/Glossary/Callback_function");
            //обычно после fetch нужно ещё преобразовать данные в json или text
            return await response.json();
        }

        //главная логика
        async function init() {
            try {
                const data = await getPageData();
                console.log("Успех!", data);
            } catch (error) {
                console.error("Ошибка загрузки: ", error);
            }
        }

        init();

    если не нужно создавать лишнюю асинхронную функцию init, то можно
    использовать промисы напрямую:

        fetch("/ru/docs/Glossary/Callback_function")
            .then(response => response.json())
            .then(data => {
                console.log("Данные получены: ", data);
            })
            .catch(err => console.error("Произошла ошибка: ", err));

у eventListner есть 3 параметра:

    1-ый аргумент (триггер):
        - 'click' - обычный клик мыши;
        - 'dblclick' - двойной щелчок;
        - 'mousedown' / 'mouseup' - кнопку зажали или отпустили;
        - 'mousemove' - движение курсора над элементом;
        - 'mouseenter' / 'mouseleave' - курсор вошёл в границы элемента или покинул;
        - 'keydown' - клавиша нажата (работает пока зажата);
        - 'keyup' - клавишу отпустили;
        - 'submit' - отправка всей формы (вешается на <form>, а не на кнопку);
        - 'input' - значение в поле ввода изменилось (срабатывает на каждый введённый
        символ);
        - 'change' - поле потеряло фокус после изменения (или выбран пункт в
        select. для выпадающих списков событие 'change' срабатывает сразу, как
        только пользователь выбрал новый пункт. это основной способ узнать, что
        именно выбрал юзер);
        - 'focus' / 'blur' - поле ввода получило или потеряло фокус;
        - 'DOMContentLoaded' - (на объекте document) HTML полностью загружен и дерево
        DOM построено;
        - 'resize' - изменился размер окна браузера;
        - 'scroll' - пользователь прокрутил страницу или элемент;
        для мобильных: 'touchstart', 'touchend', 'touchmove';

        в HTML-атрибутах пишем с префиксом "on" - "onclick", но в addEventListener
        всегда без - "click";
        можно создавать кастомные события и вызывать их вручную через dispatchEvent;

        чтобы отследить конкретную клавишу, нужно использовать объект события event,
        который автоматически попадает во второй аргумент. у него есть свойство .key
        (название клавиши) или .code (физическое место клавиши). пример кода:

            //обычно слушаем всё окно (window) или конкретный input
            window.addEventListener('keydown', (event) => {
                if (event.key === 'Enter') {
                    console.log('Вы нажали Enter!');
                }

                if (event.code === 'Space') {
                    console.log('Нажат Пробел');
                }
            });

        создать объект события можно через new Event() или new CustomEvent(),
        а затем вызвать dispatchEvent:

            const btn = document.querySelector('button');

            //1. настраиваем обычный слушатель
            btn.addEventListener('click', () => {
                alert('Клик сработал!');
            });

            //2. имитируем клик программно (например, через 2 секунды)
            setTimeout(() => {
                const myClick = new Event('click'); //создаём объект события
                btn.dispatchEvent(myClick);         //кидаем событие в элемент
            }, 2000);

        можно передавать свои данные через свойство detail:

            cinst myEvent = new CustomEvent('userLogin', {
                detail: {name: 'Ivan'}
            });

            window.dispatchEvent(myEvent);

        CustomEvent - это отличный способ организовать общение между разными
        частями приложения, которые ничего не знают друг о друге напрямую. ситуация:
            один скрипт, который отвечает за логику входа пользователя, и другой
            скрипт, который обновляет виджет пользователя в апке сайта. пример -
            передача данных о пользователе:

                1. логика входа (здесь создаём событие и прикрепяем к нему данные
                в объекте detail):

                    //функция, которая имитирует вход в систему
                    function loginUser() {
                        const userData = { id: 1, name: 'Максим', status: 'online'};

                        //создаём кастомное событие
                        const loginEvent = new CustomEvent('userLoggedIn', {
                            detail: userData,   //передаём объект с данными
                            bubbles: true       //позволяем событию всплывать вверх по DOM
                        });

                        //отправляем событие от лица документа или конкретного элемента
                        document.dispatchEvent(loginEvent);
                    }

                2. виджет в шапке (этот скрипт просто слушает документ и ждёт, когда
                прилетит событие):
                    
                    document.addEventListener('userLoggedIn', (event) => {
                        //данные лежат внутри свойства .detail
                        const user = event.detail;

                        const header = document.querySelector('#header-user-name');
                        header.textContent = `Привет, ${user.name}!';

                        console.log(`Пользователь ${user.name} вошёл в систему.`);
                    });

                в чём же главные плюсы такого подхода:

                    - независимость (Decoupling): скрипту логики не важно, есть ли
                    на странице виджет шапки. его задача оповестить всё пространство;
                    - много получателей: это событие могут одновременно услышать
                    сколько угодно скриптов и каждый может отреагировать по-своему;
                    - чистота: не нужно создавать глобальные переменные или сложные
                    цепочки функций;

                в настройках CustomEvent (второй аргумент) часто добавлябт
                { bubbles: true }. если этого не сделать, то событие будет видно
                только на том элементе, на котором оно было вызвано. с bubbles: true
                оно поднимается до самого window, и его можно будет поймать где угодно;

                третий аргумент может быть либо useCapture, либо options:

                    useCapture:
                        либо true, либо false. меняет поведение перехвата событий,
                        отчего меняется очерёдность выпонения;
                        когда проводится клик по условной кнопке, то событие проходит
                        от корня DOM до самой кнопки, и, если по пути есть событие с
                        третьим аргументом = true, то оно выполнится первым. вот пример:

                            parent.addEventListener('click', () =>
                                console.log('Родитель (Capture)'), true);
                            parent.addEventListener('click', () =>
                                console.log('Родитель (Bubble)'), false);
                            button.addEventListener('click', () =>
                                console.log('КНОПКА'));

                            в итоге при нажатии на кнопку вывод будет следующий:

                                - Родитель (Capture);
                                - КНОПКА;
                                - Родитель (Bubble);

                    в {...} можно передать несколько настроек сразу:

                        once: true - обработчик сработает ровно 1 раз и сам
                        удалится. удобно для кнопки "отправить";

                        passive: true - говорит браузеру, что вы не собираетесь
                        вызвать preventDefault(). это ускоряет прокрутку на
                        модильных устройствах;

                        capture: true/false - то же самое, что и useCapture;

                        signal - позволяет удалить обработчик через AbortController;
                    
                    preventDefault() - это метод объекта события, который говорит браузеру
                    не делать того, что он делает по умолчанию, например:
                        - у ссылки действие по умолчанию - переход на другой сайт;
                        - у формы - перезагрузка страницы при отправке;
                        - у колеса мыши или тачпада - прокрутка страницы;

                    почему это замедляет прокрутку на мобильных устройствах? потому что
                    при касании экрана и ведении пальца (события touchstart и touchmove),
                    браузер не знает заранее, собирается пользователь прокрутить страницу
                    или скрипт хочет отменить эту прокрутку через preventDefault():

                        - ожидание: браузер ставит поток рендеригна на паузу и ждёт,
                        пока выполнится js-код;
                        - проверка: он смотрит: вызван ли preventDefault() или нет;
                        - результат: если не вызван, то браузер, наконец, начинает
                        двигать страницу;
                    
                    из-за этого и появляется небольшая задержка прокрутки. сейчас часто
                    по умолчанию ставят passive: true - это обещание браузеру, что
                    в коде не будет вызываться preventDefault() и он может начать скроллить
                    без поиска;

для парсирования из текста в числа лучше использовать Number(), чем parseInt();
